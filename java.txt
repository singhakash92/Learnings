Autoboxing is automatic conversion of primitives to wrapper objects, and unboxing is the reverse — both done by the compiler to bridge primitives and the object-oriented world of Java.
int num = 5;
Integer boxed = num;  // Autoboxing: int → Integer (compiler does Integer.valueOf(num))

Integer boxed = Integer.valueOf(10);
int num = boxed;  // Unboxing: Integer → int (compiler does boxed.intValue())

27) Inner class
public class Car {
    String color = "Red";        // Car's own variable
    int speed = 100;

    public class Engine {        // Non-static inner class
        int horsepower = 450;

        public void showDetails() {
            // Engine can see Car's variables directly!
            System.out.println("Car color: " + color);      // ✅ Works!
            System.out.println("Car speed: " + speed);      // ✅ Works!
            System.out.println("Horsepower: " + horsepower);
        }
    }
}

public class Main {
    public static void main(String[] args) {
        
        // STEP 1 — Create the outer object FIRST
        Car myCar = new Car();
        
        // STEP 2 — Use the outer object to create the inner object
        Because `Engine` is **non-static** — it is tied to a specific `Car` instance.
        Car.Engine myEngine = myCar.new Engine();
        
        // Now use it
        myEngine.showDetails();
    }
}

import java.util.*;
import java.util.stream.*;

public class IteratorExamples {
    public static void main(String[] args) {

        // ── 1. FOR-EACH ──────────────────────────────────────
        List<String> fruits = List.of("Apple", "Banana", "Mango");
        for (String f : fruits) {
            System.out.println(f);          // Apple, Banana, Mango
        }

        // ── 2. ITERATOR ──────────────────────────────────────
        List<Integer> nums = new ArrayList<>(List.of(1,2,3,4,5,6));
        Iterator<Integer> it = nums.iterator();
        while (it.hasNext()) {
            if (it.next() % 2 == 0) it.remove(); // remove evens
        }
        System.out.println(nums);           // [1, 3, 5]

        // ── 3. LIST ITERATOR ─────────────────────────────────
        List<String> names = new ArrayList<>(List.of("Alice","Bob","Dave"));
        ListIterator<String> lit = names.listIterator();
        while (lit.hasNext())
            lit.set(lit.next().toUpperCase()); // uppercase in place
        while (lit.hasPrevious())
            System.out.print(lit.previous() + " "); // DAVE BOB ALICE
        System.out.println();

        // ── 4. SPLITERATOR (via parallel stream) ─────────────
        int sum = List.of(1,2,3,4,5,6,7,8)
                      .parallelStream()
                      .filter(n -> n % 2 == 0)
                      .mapToInt(Integer::intValue)
                      .sum();
        System.out.println("Sum of evens: " + sum); // 20
    }
}
